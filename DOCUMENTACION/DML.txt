CREATE DATABASE ecommercegt;
CREATE TABLE rol(
nombre_rol VARCHAR(20) PRIMARY KEY);
CREATE TABLE usuario(
dpi VARCHAR(15) PRIMARY KEY,
nombre VARCHAR(75) NOT NULL,
correo VARCHAR(50) NOT NULL,
contrasenia VARCHAR(150) NOT NULL,
suspendido BOOLEAN DEFAULT FALSE,
rol VARCHAR(20) NOT NULL,
CONSTRAINT fk_rol FOREIGN KEY (rol) REFERENCES rol(nombre_rol));
CREATE TABLE producto(
id_producto SERIAL PRIMARY KEY,
nombre_producto VARCHAR(75) NOT NULL,
descripcion VARCHAR(100) NOT NULL,
precio DECIMAL(7,2) NOT NULL,
stock INT NOT NULL,
estado VARCHAR(50) NOT NULL,
categoria VARCHAR(50) NOT NULL,
aceptado BOOLEAN,
dpi_usuario VARCHAR(15) NOT NULL,
CONSTRAINT fk_dpi_usuairo FOREIGN KEY (dpi_usuario) REFERENCES usuario(dpi));
CREATE TABLE comentario(
id_comentario SERIAL PRIMARY KEY,
calificacion INT NULL,
comentario VARCHAR(250) NOT NULL,
dpi_usuario VARCHAR(15) NOT NULL,
id_producto INT NOT NULL,
CONSTRAINT fk_dpi_usuario_comenta FOREIGN KEY(dpi_usuario) REFERENCES usuario(dpi),
CONSTRAINT fk_id_producto_comentado FOREIGN KEY(id_producto) REFERENCES producto(id_producto));
CREATE TABLE compra(
id_compra SERIAL PRIMARY KEY,
id_producto INT NOT NULL,
dpi_usuario VARCHAR(15) NOT NULL,
fecha_compra DATE NOT NULL,
CONSTRAINT fk_id_producto_comprado FOREIGN KEY(id_producto) REFERENCES producto(id_producto),
CONSTRAINT fk_dpi_usuario_compra FOREIGN KEY(dpi_usuario) REFERENCES usuario(dpi));
CREATE TABLE pedido(
id_pedido SERIAL PRIMARY KEY,
fecha_llegada DATE NOT NULL,
estado_pedido VARCHAR(50) NOT NULL,
id_producto INT NOT NULL,
dpi_usuario VARCHAR(15) NOT NULL,
CONSTRAINT fk_id_producto_pedido FOREIGN KEY(id_producto) REFERENCES producto(id_producto),
CONSTRAINT fk_dpi_usuario_pedido FOREIGN KEY(dpi_usuario) REFERENCES usuario(dpi));
CREATE TABLE sancion(
id_sancion SERIAL PRIMARY KEY,
motivo VARCHAR(250) NOT NULL,
fecha_sancion DATE NOT NULL,
estado VARCHAR(50) NOT NULL,
dpi_usuario_sancionado VARCHAR(15) NOT NULL,
CONSTRAINT fk_dpi_usuario_sancionado FOREIGN KEY(dpi_usuario_sancionado) REFERENCES usuario(dpi));
CREATE TABLE notificacion(
id_notificacion SERIAL PRIMARY KEY,
correo_notificado VARCHAR(50) NOT NULL,
fecha_notificacion DATE NOT NULL);
